便签 #

Manual

文件 #

打开文件 #

(find-file "path/to/file")

使用打开链接的方式，打开指定文件 #

(org-open-file "path/to/file")

打开链接 #

C-c C-o 

(org-open-at-point)

文本 #

在光标处插入 #

(insert "Hello")

读取输入 #

(read-string "Promots: ")

搜索并替换 #

字符串replace-string #

replace-string会跳过链接匹配

全文搜索并替换 #

(while (search-forward "file:tt.txt" nil t)
  (replace-match "file:pp.txt" nil t))

字符串查找 #

search-backward #

(search-backward "=")

;; 查找成功返回t
(search-backward "[[ue:" nil t)

正则查找 #

re-search-forward #

  • 目的字符串 
    [[ue:UE_5.1/Engine/Source/Runtime/Core/Public/Math/Interval.h=
  • 正则匹配 
    (re-search-forward (rx "[[ue:" (group (0+ (not "]"))) "][虚幻引擎:" (group (0+ (not "]"))) "]]" ))

;; 匹配返回t
(re-search-forward (rx "[[ue:" (group (0+ (not "]"))) "][虚幻引擎:" (group (0+ (not "]"))) "]]" ) nil t)

复制 #

(copy-region-as-kill (point-min) (point-max))

光标 #

光标所在位置 #

(point)

去到指定位置 #

(goto-char (point))

光标信息 #

(what-cursor-position)

记忆光标位置, 处理结束恢复 #

(save-excursion
  (insert "hello"))

获取光标间字符串 #

(buffer-substring-no-properties (point-min) (point-max))

缓冲区 #

获取buffer内容 #

(buffer-string)

在临时buffer中做处理 #

(with-temp-buffer
  (insert "hello"))

保存buffer #

(save-buffer)

关闭buffer #

(kill-buffer)

获取buffer对应的文件名 #

(buffer-name)

获取给定文件的buffer #

(get-buffer (buffer-name))

控制语句 #

条件控制语句 #

if #

  • 缩进作为true和false的分支
  • 各一条语句 
(if (= 3 5)
    (message "equal")
  (message "not equal"))

when：cond为true时继续 #

(when (< 3 5)
  (setq ooout '("less"))
  (message "%s" (cons "multi" ooout)))

unless：cond为false时继续 #

(unless (> 3 4)
  (setq ooout '("false" "3 < 4" "is" "true" "!"))
  (pop ooout)
  (message "%s" ooout))

循环while #

break #

Catch-and-Throw 

(let ((cnt 0))
  (catch 'break
    (while t
      (setq cnt (+ cnt 1))
      (sleep-for 1)
      (if (= cnt 5)
          (throw 'break t)              
        )
      )
    )
  )

功能函数 #

等待 #

(sleep-for 2)

hook #

添加hook #

(add-hook 'emms-player-started-hook (lambda () (emms-seek-to "05:05") (emms-pause)))

移除hook #

(remove-hook 'emms-player-started-hook (lambda () (emms-seek-to "05:05") (emms-pause)))

函数调用 #

funcall #

第一个参数是函数，之后的组成参数列表，传入第一个参数对应的函数 

(funcall (emms-player-get player 'start) track)

;; (emms-player-get player 'start)  查dictionary得到emms-player-mpv-start
;; (emms-player-mpv-start track)

apply-partially #

  • 类似C++的bind，预设函数调用的参数
  • 使用时，根据需求传入剩余参数

  • 返回函数

    将args作为func的前n个参数传入，调用函数时，和实际传参拼接，传给func 

    (defun apply-partially (func &rest args)
  ;;...
  )

  • 定义函数

    (defun emms-player-mpv-cmd (cmd &optional handler)
  ;;...
  )

(setq func (apply-partially 'emms-player-mpv-cmd play-cmd
                            (lambda (_mpv-data mpv-error)
                              (when (eq mpv-error 'connection-error)
                                (emms-player-mpv-cmd play-cmd)))))

(funcall func)

定义变量 #

使用let定义局部变量 #

(let ((x 10) (y 20) (z 30))
  (message (format "%d %d %d" x y z)))

变量之间存在引用 #

(let* ((x 10) (y (+ x 10)))
  (message (format "%d %d" x y)))

使用setq对局部变量值进行修改 #

使用setq定义全局变量 #

可以使用 C-h v 查看变量值

buffer-local变量 #

buffer内设置变量使用setq, 所有buffer生效使用setq-default

函数 #

返回值 #

函数最后一条语句的执行结果作为返回值

定义 #

可选参数：传参时，arg1必须匹配，arg2可选 #

(defun test-func (arg1 &option arg2)
  ;;...
  )

可变参数：传参时，arg1必须匹配，之后传入参数个数可变 #

必须放在参数列表最后 

(defun test-func (arg1 &rest args)
  ;;...
  )

数据结构 #

string #

删除空格 #

(string-trim " Hello ")

切分字符串成列表 #

(split-string opath "/")

;; 不包含为空的元素
(split-string opath "/" t)

判等 #

(string= "hello" "Hello")

不改变字符串, 返回去除前缀结果 #

(string-remove-prefix "/" slot)

list #

构造list #

(setq slotlist '())

元素操作 #

  • 将元素添加到列表(头插)

    (add-to-list 'slotlist export-name)

  • 删除首元素

    (pop list)

  • 获取指定元素

    (nth N list)

list操作 #

  • 长度

    (length list)

  • 打印

    (print list)

  • 返回头插的新list

    (cons x list)

  • 返回拼接后的新list

    (append list1 list2)

  • 将list转换为字符串

    (format "%s" list)

  • 遍历列表

    (dolist (item the-list)
  ;; 操作item
  )

  • 将末尾元素组成列表

    (last list)

  • 获取首元素

    (car list)

  • 返回反转列表拷贝

    (reverse slotlist)

alist #

Associate List 

(setq list '(("mary" . 23) ("john" . 24) ("smith" . 33)))

通过key获取键值对 #

(assoc "john" list)

通过value获取键值对 #

(rassoc 24 list)

获取键值对的key #

(car (assoc "john" x))

获取键值对的value #

(cdr (assoc "john" x))
